#如果你使用 NGINX 服务器运行 ModPHP，请修改下面的代码，
#将目录修改为你的网站目录，然后修改 NGINX 服务器配置文件，
#将修改好的代码复制到 NGINX 配置文件 nginx.conf 中，然后重启 NGINX 服务器。

http {
	# BALABALA...
	server {
		#BALABALA...

		#仅修改和复制下面的代码
		#将 root 的值修改为服务器所设置的根目录
		#将 /modphp/ 修改为你的网站所在目录，如果是根目录，则设置为 /
		
		location ~ \.php($|/) {
			if (!-f $request_filename) {
                rewrite . /modphp/index.php;
            }
		    root           "D:/Program Files/wamp/www";
		    fastcgi_pass   127.0.0.1:9000;
		    fastcgi_index  index.php;
		    fastcgi_split_path_info ^(.+\.php)(.*)$;
		    fastcgi_param  PATH_INFO $fastcgi_path_info;
		    fastcgi_param  SCRIPT_FILENAME   $document_root$fastcgi_script_name;
		    include        fastcgi_params;
		}

		location /modphp/ {
		    root   "D:/Program Files/wamp/www";
		    index  index.php index.html index.htm;
		    if (!-f $request_filename) {
		        rewrite . /modphp/index.php;
		    }
		}

		#将修改上面的代码，并将其复制到 NGINX 配置文件的相同结构内
	}
}

#修改完配置文件后，重启 NGINX 服务器，即可使用 ModPHP。
#PHP Fast-CGI 启动命令类似下面这样，在控制台中输入：
# php-cgi -b 127.0.0.1:9000
